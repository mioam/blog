### 题目大意

> 有$n$个团伙，每个团伙有$s_i$个人，团伙之间连成一张竞赛图（任意两点间有且仅由一条有向边）。一开始知道一些人有真的金块。接下来$10^{100}$年间，对每一条边$u$->$v$，第$i$小时时，第$u$个团伙的第$i\%s_u$个人会给第$v$个团伙的第$i\%s_v$个人一个假的金块，当且仅当前者有金块且后者没有。
> 现在定义一个团伙的权值为真的金块个数$~$真的金块个数+假的金块个数，将团伙按权值从大到小排名相等的可以任意排列。问有多少大小为$b$的团伙集合使得他们的排名可以都在前$a$。
>
> $n\leq5000 ,\sum s_i\leq2*10^6$


### 题解

考虑将问题分为两个部分。首先考虑竞赛图的性质：

 - 一个强连通的竞赛图有至少一条曼哈顿回路
 - 一个无环的竞赛图仅有一种拓扑序

那么一个显然的想法是使用tarjan缩点，同一个强联通分量缩成一个有$gcd$个人的点。接下来在DAG上沿拓扑序传递金块即可。

对于集合的统计，只需要枚举选中的$b$个团伙中权值最小的，然后枚举选择的权值最大值大于他但是最小值小于他的团伙的个数，使用预处理组合数的方式可以在$O(n^2+\sum s_i)$完成。

### 实现

```
#include<bits/stdc++.h>
#define P 1000000007
#define ll long long
#define N 5005
using namespace std;
int n,a,b,s[N],sz[N],C[N][N],Ans;char c[2000005];
short to[N*N],fr[N*N];int nxt[N*N],fst[N],l;
int st[N],top,bl[N],cnt,dfn[N],low[N],clk;
vector<short>p[N],v[N],e[N];
struct T
{
	int x,y;
}d[N];
int cmp(T a,T b)
{
	return a.y==b.y?a.x>b.x:a.y>b.y;
}
void link(int x,int y)
{
	to[++l]=y;fr[l]=x;nxt[l]=fst[x];fst[x]=l;
}
void dfs(int x)
{
	low[x]=dfn[x]=++clk;
	st[++top]=x;
	for (int i=fst[x];i;i=nxt[i])
		if (!dfn[to[i]])
		{
			dfs(to[i]);
			low[x]=min(low[x],low[to[i]]);
		}
		else if (!bl[to[i]])
			low[x]=min(low[x],low[to[i]]);
	
	if (low[x]==dfn[x])
	{
		cnt++;
		while(st[top]!=x)
			bl[st[top--]]=cnt;
		bl[st[top--]]=cnt;
	}
}
short _to[N*N];int _nxt[N*N],_fst[N],_l,rd[N];
void lk(int x,int y)
{
	_to[++_l]=y;_nxt[_l]=_fst[x];_fst[x]=_l;rd[y]++;
}
vector<short>tp;
int tmp;
void work(int x)
{
	rd[x]=-1;tmp=x;
	for (int i=_fst[x];i;i=_nxt[i])
	{
		rd[_to[i]]--;
		if (rd[_to[i]]==0)
		{
			int g=__gcd(sz[tmp],sz[_to[i]]);
			tp.clear();
			for (int j=0;j<g;j++)tp.push_back(0);
			for (int j=0;j<sz[tmp];j++)
				if (e[tmp][j]) tp[j%g]=1;
			for (int j=0;j<sz[_to[i]];j++)
				if (tp[j%g]) e[_to[i]][j]=1;
			
			
			work(_to[i]);
		}
	}
}
int bit[2000005];
void add(int x)
{
	x++;
	for (;x<=2000002;x+=x&-x)
		bit[x]++;
}
int qry(int x)
{
	x++;
	int ans=0;
	for (;x;x-=x&-x)
		ans+=bit[x];
	return ans;
}
int main()
{
	scanf("%d%d%d",&n,&a,&b);
	for (int i=1;i<=n;i++)
	{
		scanf("%s",c+1);
		for (int j=1;j<=n;j++)
			if (c[j]=='1')
				link(i,j);
	}
	
	for (int i=1;i<=n;i++)
		if (!dfn[i])
			dfs(i);
	
	int Mx=0;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&s[i]);Mx=max(Mx,s[i]);
		sz[bl[i]]=__gcd(sz[bl[i]],s[i]);
		p[bl[i]].push_back(i);
		scanf("%s",c+1);
		for (int j=1;j<=s[i];j++)
		{
			v[i].push_back(c[j]=='1');
			d[i].x+=c[j]=='1';
		}
	}
	for (int i=1;i<=cnt;i++)
	{
		for (int j=0;j<sz[i];j++)
			e[i].push_back(0);
		for (int j=0;j<p[i].size();j++)
		{
			for (int k=0;k<v[p[i][j]].size();k++)
			if (v[p[i][j]][k])
				e[i][k%sz[i]]=1;
		}
	}
	for (int i=1;i<=l;i++)
		if (bl[fr[i]]!=bl[to[i]])
			lk(bl[fr[i]],bl[to[i]]);
	
	for (int i=1;i<=cnt;i++)
		if (rd[i]==0) work(i);
	
	for (int i=1;i<=n;i++)
	{
		for (int j=0;j<s[i];j++)
			if (v[i][j]||e[bl[i]][j%sz[bl[i]]]) d[i].y++;
	}
	sort(d+1,d+n+1,cmp);
	
	C[0][0]=1;
	for (int i=1;i<=n;i++)
	{
		C[i][0]=1;
		for (int j=1;j<=i;j++)
			C[i][j]=(C[i-1][j]+C[i-1][j-1])%P;
	}
	for (int i=1;i<=n;i++)
	{
		if (i>=b)
		{
			if (i<=a) Ans=(Ans+C[i-1][b-1])%P;
			else
			{
				int u=qry(d[i].y),v=i-1-u;
				for (int j=0;j<=u&&j<b;j++)
					if (v+j+1<=a)
						Ans=(Ans+(ll)C[u][j]*C[v][b-j-1])%P;
			}
		}
		add(d[i].x);
	}
	printf("%d\n",Ans);
}
```
